# Provider Setup

Set up the Chakra UI Provider at the application root.

```tsx
// 1. Install
// npm i @chakra-ui/react @emotion/react
// npx @chakra-ui/cli snippet add

// 2. app/layout.tsx (Next.js App Router)
import { Provider } from "@/components/ui/provider"

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <Provider>{children}</Provider>
      </body>
    </html>
  )
}

// 3. Use Chakra components anywhere in the tree
import { Button, HStack } from "@chakra-ui/react"

export default function Page() {
  return (
    <HStack>
      <Button>Save</Button>
      <Button variant="outline">Cancel</Button>
    </HStack>
  )
}
```

## Notes

- `Provider` (generated by CLI snippet) composes `ChakraProvider` and `ThemeProvider` from `next-themes`
- Add `suppressHydrationWarning` to `<html>` to suppress color-mode hydration mismatch warnings
- For Pages Router, wrap `_app.tsx` instead of `layout.tsx`
- Minimum Node.js version: 20.x
